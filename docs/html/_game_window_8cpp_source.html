158void GameWindow::add_quit_event()
159{
160 //default event to process quit button pressing
161
162 auto pred = [](const SDL_Event& e) -> bool
163 {
164 //if user press X on window, then it's time to quit window
165 return e.type == SDL_QUIT;
166 };
167 auto fn = [&](void* state) -> void
168 {
169 //if pred returns true, then we change quit variable's value to true

172 bool* val = static_cast<bool*>(state);
173 *val = true;
174 }
175 };

177 //bind predicat, function and state together to get event
178 quit_event = new framework::KeyboardEvent(pred, fn, static_cast<void*>(&this->quit));
