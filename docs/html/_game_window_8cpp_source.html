161void GameWindow::add_quit_event()
162{
163 //default event to process quit button pressing
164
165 auto pred = [](const SDL_Event& e) -> bool
166 {
167 //if user press X on window, then it's time to quit window
168 return e.type == SDL_QUIT;
169 };
170 auto fn = [&](void* state) -> void
171 {
172 //if pred returns true, then we change quit variable's value to true
173 if (state != ZERO_STATE)
174 {
175 bool* val = static_cast<bool*>(state);
176 *val = true;
177 }
178 };
179
180 //bind predicat, function and state together to get event
181 quit_event = new framework::KeyboardEvent(pred, fn, static_cast<void*>(&this->quit));
